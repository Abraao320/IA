        <input type="text" id="user-input" placeholder="Digite sua mensagem...">
        <button onclick="enviarMensagem()">Enviar</button>
    </div>
    <div id="status">Carregando a IA... Isso pode levar um minuto na primeira vez.</div>

    <!-- Importando a biblioteca Transformers.js direto do CDN -->
    <script type="module">
        import { pipeline } from 'https://cdn.jsdelivr.net/npm/@xenova/transformers@2.17.1';

        // Usando um modelo muito leve que funciona no navegador
        const gerador = await pipeline('text-generation', 'Xenova/Qwen1.5-0.5B-Chat');
        document.getElementById('status').innerText = 'IA pronta! Pode mandar bala.';

        // Função global para o botão
        window.enviarMensagem = async function() {
            const input = document.getElementById('user-input');
            const texto = input.value;
            if (!texto) return;

            adicionarNoChat('user', texto);
            input.value = '';
            document.getElementById('status').innerText = 'Pensando...';

            try {
                // Gera a resposta localmente no navegador
                const resultado = await gerador(texto, { max_new_tokens: 100, temperature: 0.7 });
                const resposta = resultado[0].generated_text;
                
                adicionarNoChat('bot', resposta);
            } catch (e) {
                adicionarNoChat('bot', 'Erro ao gerar resposta.');
                console.error(e);
            }
            document.getElementById('status').innerText = 'IA pronta.';
        }

        function adicionarNoChat(tipo, texto) {
            const chatBox = document.getElementById('chat-box');
            const div = document.createElement('div');
            div.className = `msg ${tipo}`;
            div.innerText = texto;
            chatBox.appendChild(div);
            chatBox.scrollTop = chatBox.scrollHeight;
        }
    </script>
